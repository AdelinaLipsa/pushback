    ```typescript
async function handleAnalyze() {
  if (!messageInput.trim()) return
  if (isAtLimit) { setShowUpgrade(true); return }
  setAnalyzeLoading(true)
  setAnalyzeError(null)
  setAnalysisResult(null)

  const res = await fetch(`/api/projects/${projectId}/analyze-message`, {
    method: 'POST',
    headers: { 'Content-Type': 'application/json' },
    body: JSON.stringify({ message: messageInput }),
  })

  const data = await res.json()
  setAnalyzeLoading(false)

  if (res.status === 403 && data.error === 'UPGRADE_REQUIRED') {
    setShowUpgrade(true)
    return
  }

  if (!res.ok) {
    setAnalyzeError(data?.error ?? 'Something went wrong. Please try again.')
    return
  }

  setAnalysisResult(data)
  const matchedTool = DEFENSE_TOOLS.find(t => t.type === data.tool_type)
  if (matchedTool) {
    selectTool(matchedTool)
  }
}
    ```

<threat_model>
## Trust Boundaries

| Boundary | Description |
|----------|-------------|
| user textarea → handleAnalyze fetch body | User-controlled `messageInput` string is serialized into JSON and POSTed; sanitization happens server-side via Zod in the route |
| API response → analysisResult state | The `{ tool_type, explanation, situation_context }` values from the route are rendered directly into JSX — potential XSS if React escaping is bypassed |

## STRIDE Threat Register

| Threat ID | Category | Component | Disposition | Mitigation Plan |
|-----------|----------|-----------|-------------|-----------------|
| T-06-03-01 | Tampering / XSS | `analysisResult.explanation` and `situation_context` rendered in JSX | accept | React JSX text nodes auto-escape HTML entities. Values are rendered as `{analysisResult.explanation}` (text node), not via `dangerouslySetInnerHTML`. No additional escaping needed. |
| T-06-03-02 | Tampering / Prompt Injection | `messageInput` value sent to API | mitigate | Route validates `message` with Zod `min(10).max(5000)`. UI enforces `maxLength={3000}` as an additional UX guard. Server is authoritative. |
| T-06-03-03 | Elevation of Privilege / Credit Abuse | isAtLimit bypass in UI | accept | `isAtLimit` check in `handleAnalyze` is a UX convenience; the server RPC gate is the authoritative enforcement (DETECT-03). Bypassing the UI check only reaches the API, which will return 403 UPGRADE_REQUIRED. |
  | T-06-03-04 | Denial of Service / Rapid clicks | Double-submit on Analyze button | mitigate | Button is `disabled={analyzeLoading}` and `opacity: 0.6, cursor: 'not-allowed'` during loading. Race between two fast clicks is blocked at the server by the atomic RPC gate. |
</threat_model>
