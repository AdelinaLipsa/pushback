3. Add the handleCopyPushback function inside the component, after the existing useState declarations:
   ```typescript
   async function handleCopyPushback() {
     try {
       await navigator.clipboard.writeText(clause.pushback_language)
       setCopied(true)
       setTimeout(() => setCopied(false), 2000)
     } catch {
       // clipboard unavailable — fail silently (A3 from RESEARCH.md)
     }
   }
   ```
