```typescript
'use client'

import { useState } from 'react'
import { DefenseResponse, Plan } from '@/types'
import CopyButton from '@/components/shared/CopyButton'

interface ResponseHistoryProps {
  responses: DefenseResponse[]
  plan: Plan
}

const TOOL_LABELS: Record<string, string> = {
  scope_change: 'Scope Change',
  payment_first: 'Payment Reminder',
  payment_second: 'Payment Follow-Up',
  payment_final: 'Final Payment Notice',
  revision_limit: 'Revision Limit',
  kill_fee: 'Kill Fee',
  delivery_signoff: 'Delivery Sign-Off',
  dispute_response: 'Dispute Response',
}

export default function ResponseHistory({ responses, plan }: ResponseHistoryProps) {
  const [expanded, setExpanded] = useState<string | null>(null)
  const [upgradeLoading, setUpgradeLoading] = useState(false)

  async function handleUpgrade() {
    setUpgradeLoading(true)
    const res = await fetch('/api/checkout', { method: 'POST' })
    const data = await res.json()
    if (data.url) {
      window.location.href = data.url
    } else {
      setUpgradeLoading(false)
    }
  }

  if (responses.length === 0) {
    return (
      <p style={{ color: 'var(--text-muted)', fontSize: '0.9rem' }}>No messages generated yet.</p>
    )
  }

  return (
    <div style={{ display: 'flex', flexDirection: 'column', gap: '0.75rem' }}>
      {responses.map((r, index) => {
        const isLocked = plan === 'free' && index >= 3

        return (
          <div
            key={r.id}
            style={{
              position: 'relative',
              backgroundColor: 'var(--bg-surface)',
              border: '1px solid var(--bg-border)',
              borderRadius: '0.75rem',
              overflow: 'hidden',
            }}
          >
            {/* Card content — blurred when locked (D-04) */}
            <div style={isLocked ? { filter: 'blur(4px)', pointerEvents: 'none', userSelect: 'none' } : undefined}>
              <button
                onClick={() => setExpanded(expanded === r.id ? null : r.id)}
                style={{
                  width: '100%', display: 'flex', alignItems: 'center', justifyContent: 'space-between',
                  padding: '1rem 1.25rem', background: 'none', border: 'none', cursor: 'pointer',
                  color: 'var(--text-primary)', textAlign: 'left',
                }}
              >
                <div>
                  <div style={{ fontWeight: 500, fontSize: '0.9rem', marginBottom: '0.2rem' }}>
                    {TOOL_LABELS[r.tool_type] ?? r.tool_type}
                  </div>
                  <div style={{ color: 'var(--text-muted)', fontSize: '0.75rem' }}>
                    {new Date(r.created_at).toLocaleDateString()} · {r.was_sent ? '✓ Sent' : r.was_copied ? '✓ Copied' : 'Not sent'}
                  </div>
                </div>
                <span style={{ color: 'var(--text-muted)', fontSize: '0.8rem' }}>{expanded === r.id ? '▲' : '▼'}</span>
              </button>

              {expanded === r.id && (
                <div style={{ padding: '0 1.25rem 1.25rem' }}>
                  <pre style={{
                    fontFamily: 'var(--font-mono), monospace', fontSize: '0.8rem', lineHeight: 1.7,
                    color: 'var(--text-primary)', whiteSpace: 'pre-wrap', wordBreak: 'break-word',
                    backgroundColor: 'var(--bg-base)', border: '1px solid var(--bg-border)',
                    borderRadius: '0.5rem', padding: '1rem', margin: '0 0 1rem',
                  }}>
                    {r.response}
                  </pre>
                  <CopyButton text={r.response} responseId={r.id} />
                </div>
              )}
            </div>

            {/* Upgrade overlay — only shown for locked cards (D-05) */}
            {isLocked && (
              <div style={{
                position: 'absolute',
                inset: 0,
                display: 'flex',
                flexDirection: 'column',
                alignItems: 'center',
                justifyContent: 'center',
                backgroundColor: 'rgba(0,0,0,0.6)',
                gap: '0.75rem',
              }}>
                <span style={{ color: 'var(--text-secondary)', fontSize: '0.85rem' }}>
                  Upgrade to see full history
                </span>
                <button
                  onClick={handleUpgrade}
                  disabled={upgradeLoading}
                  style={{
                    backgroundColor: 'var(--brand-amber)',
                    color: '#0a0a0a',
                    fontWeight: 700,
                    padding: '0.7rem 1.5rem',
                    borderRadius: '0.5rem',
                    border: 'none',
                    cursor: upgradeLoading ? 'not-allowed' : 'pointer',
                    fontSize: '0.9rem',
                    opacity: upgradeLoading ? 0.7 : 1,
                  }}
                >
                  {upgradeLoading ? 'Loading…' : 'Upgrade to Pro'}
                </button>
              </div>
            )}
          </div>
        )
      })}
    </div>
  )
}
```

Key implementation details:
- `position: 'relative'` on the outer card div is required for the `position: 'absolute', inset: 0` overlay to work
- `overflow: 'hidden'` on the card clips the overlay to the card bounds (matches the existing style)

- The `isLocked` boolean is `plan === 'free' && index >= 3` — pro users see everything, free users see first 3 (indices 0, 1, 2) without lock
- `handleUpgrade` matches the exact pattern from `components/shared/UpgradePrompt.tsx`
- No semicolons, single quotes, `'use client'` is the very first line

<threat_model>
## Trust Boundaries

| Boundary | Description |
|----------|-------------|
| Server → Client | plan value fetched server-side from user_profiles, passed as prop — not from client |
| Client → /api/checkout POST | handleUpgrade makes an authenticated request to initiate Creem checkout |

## STRIDE Threat Register

| Threat ID | Category | Component | Disposition | Mitigation Plan |
|-----------|----------|-----------|-------------|-----------------|
| T-06-01 | Elevation of Privilege | Client-side blur bypass | accept | The blur is a UI affordance, not a security control; a determined user could inspect DOM. The actual enforcement is the RPC gate on the API routes — history is cosmetic gating |
| T-06-02 | Information Disclosure | All responses fetched from DB regardless of plan | accept | The history page is server-rendered and authenticated; the blur only hides from casual view. A user could theoretically read the raw network response. For v1, cosmetic gating is acceptable; v2 can add server-side .limit(3) for free users |
| T-06-03 | Tampering | plan prop could be overridden client-side | accept | The plan value flows from server-fetched user_profiles through a Server Component prop — there is no client-side mutation path for this prop before render |
| T-06-04 | Denial of Service | /api/checkout POST from locked card button | accept | Checkout requests are authenticated; Creem API handles rate limiting |
</threat_model>
