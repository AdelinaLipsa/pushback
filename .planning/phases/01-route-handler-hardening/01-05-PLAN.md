```typescript
export async function POST(request: Request) {
  const supabase = await createServerSupabaseClient()
  const { data: { user } } = await supabase.auth.getUser()
  if (!user) return Response.json({ error: 'Unauthorized' }, { status: 401 })

  const body = await request.json()
  const parsed = projectSchema.safeParse(body)
  if (!parsed.success) {
    const issue = parsed.error.issues[0]
    const field = String(issue.path[0])
    return Response.json({ error: `${field} is invalid: ${issue.message}` }, { status: 400 })
  }
  const { title, client_name, client_email, project_value, currency, notes } = parsed.data

  const { data, error } = await supabase
    .from('projects')
    .insert({ user_id: user.id, title, client_name, client_email, project_value, currency: currency ?? 'EUR', notes })
    .select()
    .single()

  if (error) return Response.json({ error: error.message }, { status: 500 })
  return Response.json({ project: data })
}
```

Conventions: no semicolons, single quotes.

Note: `z.number().positive()` rejects zero and negative values for project_value. `z.string().email()` validates email format with Zod's built-in email validator. Both are per D-08.

| Threat ID | Category | Component | Disposition | Mitigation Plan |

| T-05-04 | Tampering | currency arbitrary string injection | mitigate | `z.enum(['EUR', 'USD', 'GBP', 'AUD', 'CAD'])` rejects unknown currency codes; only allowed values reach the DB |
