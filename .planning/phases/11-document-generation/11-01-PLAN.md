```typescript
// ─── Documents ──────────────────────────────────────────────────────────────

type DocumentData = { document: string }

export type DocumentResult = { upgradeRequired: true } | (DocumentData & { upgradeRequired?: false }) | null

export async function generateDocument(
  projectId: string,
  body: { document_type: 'sow_amendment' | 'kill_fee_invoice' | 'dispute_package'; context?: string },
): Promise<DocumentResult> {
  try {
    const res = await fetch(`/api/projects/${projectId}/document`, {
      method: 'POST',
      headers: { 'Content-Type': 'application/json' },
      body: JSON.stringify(body),
    })
    const data = await res.json()
    if (res.status === 403 && data.error === 'PRO_REQUIRED') return { upgradeRequired: true }
    if (!res.ok) { toast.error(data?.error ?? 'Something went wrong.'); return null }
    return data as DocumentData
  } catch {
    toast.error('Network error — check your connection.')
    return null
  }
}
```

Note: error string is `'PRO_REQUIRED'` (not `'UPGRADE_REQUIRED'` which generateDefense uses) — per CONTEXT.md D-02.

The body parameter inlines the literal union for `document_type` instead of importing `DocumentType` from `@/types` — this matches the inline-literal style already used in generateDefense (which uses `DefenseTool` import only because it's already imported for typing). Both styles are valid; inline is fine here.
